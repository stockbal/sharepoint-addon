function getAsyncData() {
	// The creation/reading of the OData model is not relevant here
	var myResult;

	oModel.read("/MyEntitySet(id='1')", {
		// will get called if the service request was successful
		success: function(oResult) {
			// set the returned result to the variable
			myResult = oResult;
		},
		// will get called if an error occurrs
		error: function(oError) {
		// print the error
		console.log(oError);
		}
	});

	// return the retrieved result from the odata service
	return myResult;
}
